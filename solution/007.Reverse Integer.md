class Solution {
    public int reverse(int x) {
        int rev=0;
        int sign=0;
        if(x<0) {
            sign=1;
            x=-1*x;
        }
        while(x!=0) {
            if((Integer.MAX_VALUE-x%10)/10<rev)
                return 0;
            rev=rev*10+x%10;
            x/=10;
        }
        if(sign==1)
            rev=-1*rev;
        return rev;
    }
}
