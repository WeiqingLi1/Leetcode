
class Solution {
    public int lengthOfLongestSubstring(String s) {

        HashSet<Character> pool = new HashSet<Character>();
        int i = 0;
        int j = -1;
        int length = s.length();
        int max = j - i + 1;
        while((j + 1) < length){
            if(pool.add(s.charAt(j + 1)) == true) {
                j++;
            } else {
                pool.remove(s.charAt(i));
                i++;
            }
            if((j - i + 1) > max){
                max = j - i + 1;
            }
        }
        
        return max;
    }
}
