class Solution {
    public int removeDuplicates(int[] nums) {
        int dis = 0;
        
        for(int i=1; i<nums.length; i++) {
            if(nums[dis]!=nums[i]) {
                nums[dis+1]=nums[i];
                dis++;
            }
            
        }
        return dis+1;
    }
}
